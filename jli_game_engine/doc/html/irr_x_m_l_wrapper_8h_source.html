76  CIrrXML_IOStreamReader(IOStream* _stream)
77  : stream (_stream)
78  , t (0)
79  {
80 
81  // Map the buffer into memory and convert it to UTF8. IrrXML provides its
82  // own conversion, which is merely a cast from uintNN_t to uint8_t. Thus,
83  // it is not suitable for our purposes and we have to do it BEFORE IrrXML
84  // gets the buffer. Sadly, this forces us to map the whole file into
85  // memory.
86 
87  data.resize(stream->FileSize());
88  stream->Read(&data[0],data.size(),1);
89 
90  // Remove null characters from the input sequence otherwise the parsing will utterly fail
91  unsigned int size = 0;
92  unsigned int size_max = data.size();
93  for(unsigned int i = 0; i < size_max; i++) {
94  if(data[i] != '\0') {
95  data[size++] = data[i];
96  }
97  }
98  data.resize(size);
99 
100  BaseImporter::ConvertToUTF8(data);
101  }
102 
103  // ----------------------------------------------------------------------------------
105  virtual ~CIrrXML_IOStreamReader() {};
106 
107  // ----------------------------------------------------------------------------------
109 
112  virtual int read(void* buffer, int sizeToRead) {
113  if(sizeToRead<0) {
114  return 0;
115  }
116  if(t+sizeToRead>data.size()) {
117  sizeToRead = data.size()-t;
118  }
119 
120  memcpy(buffer,&data.front()+t,sizeToRead);
121 
122  t += sizeToRead;
123  return sizeToRead;
124  }
125 
126  // ----------------------------------------------------------------------------------
128  virtual int getSize() {
129  return (int)data.size();
130  }
