158 enum MeshChunkId
159 {
160  M_HEADER = 0x1000,
161  // char* version : Version number check
162  M_MESH = 0x3000,
163  // bool skeletallyAnimated // important flag which affects h/w buffer policies
164  // Optional M_GEOMETRY chunk
165  M_SUBMESH = 0x4000,
166  // char* materialName
167  // bool useSharedVertices
168  // unsigned int indexCount
169  // bool indexes32Bit
170  // unsigned int* faceVertexIndices (indexCount)
171  // OR
172  // unsigned short* faceVertexIndices (indexCount)
173  // M_GEOMETRY chunk (Optional: present only if useSharedVertices = false)
174  M_SUBMESH_OPERATION = 0x4010, // optional, trilist assumed if missing
175  // unsigned short operationType
176  M_SUBMESH_BONE_ASSIGNMENT = 0x4100,
177  // Optional bone weights (repeating section)
178  // unsigned int vertexIndex;
179  // unsigned short boneIndex;
180  // float weight;
181  // Optional chunk that matches a texture name to an alias
182  // a texture alias is sent to the submesh material to use this texture name
183  // instead of the one in the texture unit with a matching alias name
184  M_SUBMESH_TEXTURE_ALIAS = 0x4200, // Repeating section
185  // char* aliasName;
186  // char* textureName;
187 
188  M_GEOMETRY = 0x5000, // NB this chunk is embedded within M_MESH and M_SUBMESH
189  // unsigned int vertexCount
190  M_GEOMETRY_VERTEX_DECLARATION = 0x5100,
191  M_GEOMETRY_VERTEX_ELEMENT = 0x5110, // Repeating section
192  // unsigned short source; // buffer bind source
193  // unsigned short type; // VertexElementType
194  // unsigned short semantic; // VertexElementSemantic
195  // unsigned short offset; // start offset in buffer in bytes
196  // unsigned short index; // index of the semantic (for colours and texture coords)
197  M_GEOMETRY_VERTEX_BUFFER = 0x5200, // Repeating section
198  // unsigned short bindIndex; // Index to bind this buffer to
199  // unsigned short vertexSize; // Per-vertex size, must agree with declaration at this index
200  M_GEOMETRY_VERTEX_BUFFER_DATA = 0x5210,
201  // raw buffer data
202  M_MESH_SKELETON_LINK = 0x6000,
203  // Optional link to skeleton
204  // char* skeletonName : name of .skeleton to use
205  M_MESH_BONE_ASSIGNMENT = 0x7000,
206  // Optional bone weights (repeating section)
207  // unsigned int vertexIndex;
208  // unsigned short boneIndex;
209  // float weight;
210  M_MESH_LOD = 0x8000,
211  // Optional LOD information
212  // string strategyName;
213  // unsigned short numLevels;
214  // bool manual; (true for manual alternate meshes, false for generated)
215  M_MESH_LOD_USAGE = 0x8100,
216  // Repeating section, ordered in increasing depth
217  // NB LOD 0 (full detail from 0 depth) is omitted
218  // LOD value - this is a distance, a pixel count etc, based on strategy
219  // float lodValue;
220  M_MESH_LOD_MANUAL = 0x8110,
221  // Required if M_MESH_LOD section manual = true
222  // String manualMeshName;
223  M_MESH_LOD_GENERATED = 0x8120,
224  // Required if M_MESH_LOD section manual = false
225  // Repeating section (1 per submesh)
226  // unsigned int indexCount;
227  // bool indexes32Bit
228  // unsigned short* faceIndexes; (indexCount)
229  // OR
230  // unsigned int* faceIndexes; (indexCount)
231  M_MESH_BOUNDS = 0x9000,
232  // float minx, miny, minz
233  // float maxx, maxy, maxz
234  // float radius
235 
236  // Added By DrEvil
237  // optional chunk that contains a table of submesh indexes and the names of
238  // the sub-meshes.
239  M_SUBMESH_NAME_TABLE = 0xA000,
240  // Subchunks of the name table. Each chunk contains an index & string
241  M_SUBMESH_NAME_TABLE_ELEMENT = 0xA100,
242  // short index
243  // char* name
244  // Optional chunk which stores precomputed edge data
245  M_EDGE_LISTS = 0xB000,
246  // Each LOD has a separate edge list
247  M_EDGE_LIST_LOD = 0xB100,
248  // unsigned short lodIndex
249  // bool isManual // If manual, no edge data here, loaded from manual mesh
250  // bool isClosed
251  // unsigned long numTriangles
252  // unsigned long numEdgeGroups
253  // Triangle* triangleList
254  // unsigned long indexSet
255  // unsigned long vertexSet
256  // unsigned long vertIndex[3]
257  // unsigned long sharedVertIndex[3]
258  // float normal[4]
259 
260  M_EDGE_GROUP = 0xB110,
261  // unsigned long vertexSet
262  // unsigned long triStart
263  // unsigned long triCount
264  // unsigned long numEdges
265  // Edge* edgeList
266  // unsigned long triIndex[2]
267  // unsigned long vertIndex[2]
268  // unsigned long sharedVertIndex[2]
269  // bool degenerate
270  // Optional poses section, referred to by pose keyframes
271  M_POSES = 0xC000,
272  M_POSE = 0xC100,
273  // char* name (may be blank)
274  // unsigned short target // 0 for shared geometry,
275  // 1+ for submesh index + 1
276  // bool includesNormals [1.8+]
277  M_POSE_VERTEX = 0xC111,
278  // unsigned long vertexIndex
279  // float xoffset, yoffset, zoffset
280  // float xnormal, ynormal, znormal (optional, 1.8+)
281  // Optional vertex animation chunk
282  M_ANIMATIONS = 0xD000,
283  M_ANIMATION = 0xD100,
284  // char* name
285  // float length
286  M_ANIMATION_BASEINFO = 0xD105,
287  // [Optional] base keyframe information (pose animation only)
288  // char* baseAnimationName (blank for self)
289  // float baseKeyFrameTime
290  M_ANIMATION_TRACK = 0xD110,
291  // unsigned short type // 1 == morph, 2 == pose
292  // unsigned short target // 0 for shared geometry,
293  // 1+ for submesh index + 1
294  M_ANIMATION_MORPH_KEYFRAME = 0xD111,
295  // float time
296  // bool includesNormals [1.8+]
297  // float x,y,z // repeat by number of vertices in original geometry
298  M_ANIMATION_POSE_KEYFRAME = 0xD112,
299  // float time
300  M_ANIMATION_POSE_REF = 0xD113, // repeat for number of referenced poses
301  // unsigned short poseIndex
302  // float influence
303  // Optional submesh extreme vertex list chink
304  M_TABLE_EXTREMES = 0xE000,
305  // unsigned short submesh_index;
306  // float extremes [n_extremes][3];
307 };

351 enum SkeletonChunkId
352 {
353  SKELETON_HEADER = 0x1000,
354  // char* version : Version number check
355  SKELETON_BLENDMODE = 0x1010, // optional
356  // unsigned short blendmode : SkeletonAnimationBlendMode
357  SKELETON_BONE = 0x2000,
358  // Repeating section defining each bone in the system.
359  // Bones are assigned indexes automatically based on their order of declaration
360  // starting with 0.
361  // char* name : name of the bone
362  // unsigned short handle : handle of the bone, should be contiguous & start at 0
363  // Vector3 position : position of this bone relative to parent
364  // Quaternion orientation : orientation of this bone relative to parent
365  // Vector3 scale : scale of this bone relative to parent
366  SKELETON_BONE_PARENT = 0x3000,
367  // Record of the parent of a single bone, used to build the node tree
368  // Repeating section, listed in Bone Index order, one per Bone
369  // unsigned short handle : child bone
370  // unsigned short parentHandle : parent bone
371  SKELETON_ANIMATION = 0x4000,
372  // A single animation for this skeleton
373  // char* name : Name of the animation
374  // float length : Length of the animation in seconds
375  SKELETON_ANIMATION_BASEINFO = 0x4010,
376  // [Optional] base keyframe information
377  // char* baseAnimationName (blank for self)
378  // float baseKeyFrameTime
379  SKELETON_ANIMATION_TRACK = 0x4100,
380  // A single animation track (relates to a single bone)
381  // Repeating section (within SKELETON_ANIMATION)
382  // unsigned short boneIndex : Index of bone to apply to
383  SKELETON_ANIMATION_TRACK_KEYFRAME = 0x4110,
384  // A single keyframe within the track
385  // Repeating section
386  // float time : The time position (seconds)
387  // Quaternion rotate : Rotation to apply at this keyframe
388  // Vector3 translate : Translation to apply at this keyframe
389  // Vector3 scale : Scale to apply at this keyframe
390  SKELETON_ANIMATION_LINK = 0x5000
391  // Link to another skeleton, to re-use its animations
392  // char* skeletonName : name of skeleton to get animations from
393  // float scale : scale to apply to trans/scale keys
394 };
