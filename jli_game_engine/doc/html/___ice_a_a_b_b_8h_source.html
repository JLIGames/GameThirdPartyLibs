211  inline_ void Rotate(const Matrix4x4& mtx, AABB& aabb) const
213  {
214  // The three edges transformed: you can efficiently transform an X-only vector
215  // by just getting the "X" column of the matrix
216  Point vx,vy,vz;
217  mtx.GetRow(0, vx); vx *= (mMax.x - mMin.x);
218  mtx.GetRow(1, vy); vy *= (mMax.y - mMin.y);
219  mtx.GetRow(2, vz); vz *= (mMax.z - mMin.z);
220 
221  // Transform the min point
222  aabb.mMin = aabb.mMax = mMin * mtx;
223 
224  // Take the transformed min & axes and find new extents
225  // Using CPU code in the right place is faster...
226  if(IS_NEGATIVE_FLOAT(vx.x)) aabb.mMin.x += vx.x; else aabb.mMax.x += vx.x;
227  if(IS_NEGATIVE_FLOAT(vx.y)) aabb.mMin.y += vx.y; else aabb.mMax.y += vx.y;
228  if(IS_NEGATIVE_FLOAT(vx.z)) aabb.mMin.z += vx.z; else aabb.mMax.z += vx.z;
229  if(IS_NEGATIVE_FLOAT(vy.x)) aabb.mMin.x += vy.x; else aabb.mMax.x += vy.x;
230  if(IS_NEGATIVE_FLOAT(vy.y)) aabb.mMin.y += vy.y; else aabb.mMax.y += vy.y;
231  if(IS_NEGATIVE_FLOAT(vy.z)) aabb.mMin.z += vy.z; else aabb.mMax.z += vy.z;
232  if(IS_NEGATIVE_FLOAT(vz.x)) aabb.mMin.x += vz.x; else aabb.mMax.x += vz.x;
233  if(IS_NEGATIVE_FLOAT(vz.y)) aabb.mMin.y += vz.y; else aabb.mMax.y += vz.y;
234  if(IS_NEGATIVE_FLOAT(vz.z)) aabb.mMin.z += vz.z; else aabb.mMax.z += vz.z;
235  }

440  inline_ void Rotate(const Matrix4x4& mtx, AABB& aabb) const
442  {
443  // Compute new center
444  aabb.mCenter = mCenter * mtx;
445 
446  // Compute new extents. FPU code & CPU code have been interleaved for improved performance.
447  Point Ex(mtx.m[0][0] * mExtents.x, mtx.m[0][1] * mExtents.x, mtx.m[0][2] * mExtents.x);
448  IR(Ex.x)&=0x7fffffff; IR(Ex.y)&=0x7fffffff; IR(Ex.z)&=0x7fffffff;
449 
450  Point Ey(mtx.m[1][0] * mExtents.y, mtx.m[1][1] * mExtents.y, mtx.m[1][2] * mExtents.y);
451  IR(Ey.x)&=0x7fffffff; IR(Ey.y)&=0x7fffffff; IR(Ey.z)&=0x7fffffff;
452 
453  Point Ez(mtx.m[2][0] * mExtents.z, mtx.m[2][1] * mExtents.z, mtx.m[2][2] * mExtents.z);
454  IR(Ez.x)&=0x7fffffff; IR(Ez.y)&=0x7fffffff; IR(Ez.z)&=0x7fffffff;
455 
456  aabb.mExtents.x = Ex.x + Ey.x + Ez.x;
457  aabb.mExtents.y = Ex.y + Ey.y + Ez.y;
458  aabb.mExtents.z = Ex.z + Ey.z + Ez.z;
