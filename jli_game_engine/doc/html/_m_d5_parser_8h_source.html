251 // Convert a quaternion to its usual representation
252 inline void ConvertQuaternion (const aiVector3D& in, aiQuaternion& out) {
253 
254  out.x = in.x;
255  out.y = in.y;
256  out.z = in.z;
257 
258  const float t = 1.0f - (in.x*in.x) - (in.y*in.y) - (in.z*in.z);
259 
260  if (t < 0.0f)
261  out.w = 0.0f;
262  else out.w = sqrt (t);
263 }
