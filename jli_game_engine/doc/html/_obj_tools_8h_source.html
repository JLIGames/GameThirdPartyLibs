234 template<class string_type>
235 unsigned int tokenize( const string_type& str, std::vector<string_type>& tokens,
236  const string_type& delimiters )
237 {
238  // Skip delimiters at beginning.
239  typename string_type::size_type lastPos = str.find_first_not_of( delimiters, 0 );
240 
241  // Find first "non-delimiter".
242  typename string_type::size_type pos = str.find_first_of( delimiters, lastPos );
243  while ( string_type::npos != pos || string_type::npos != lastPos )
244  {
245  // Found a token, add it to the vector.
246  string_type tmp = str.substr(lastPos, pos - lastPos);
247  if ( !tmp.empty() && ' ' != tmp[ 0 ] )
248  tokens.push_back( tmp );
249 
250  // Skip delimiters. Note the "not_of"
251  lastPos = str.find_first_not_of( delimiters, pos );
252 
253  // Find next "non-delimiter"
254  pos = str.find_first_of( delimiters, lastPos );
255  }
256 
257  return static_cast<unsigned int>( tokens.size() );
258 }
