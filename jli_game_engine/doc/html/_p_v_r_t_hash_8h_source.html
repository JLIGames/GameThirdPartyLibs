148 #define FNV_OFFSETBIAS 2166136261U

151  return CPVRTHash();
152 
153  CPVRTHash pvrHash;
154  unsigned char* p = (unsigned char*)pData;
155  pvrHash.m_uiHash = FNV_OFFSETBIAS;

157  {
