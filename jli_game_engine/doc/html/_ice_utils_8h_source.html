58  // The population count of a binary integer value x is the number of one bits in the value. Although many machines have
59  // single instructions for this, the single instructions are usually microcoded loops that test a bit per cycle; a log-time
60  // algorithm coded in C is often faster. The following code uses a variable-precision SWAR algorithm to perform a tree
61  // reduction adding the bits in a 32-bit value:"

64  /* 32-bit recursive reduction using SWAR...
65  but first step is mapping 2-bit values
66  into sum of 2 1-bit values in sneaky way
67  */

70  x = (((x >> 4) + x) & 0x0f0f0f0f);

306  // Compute implicit coords from an index:
307  // The idea is to get back 2D coords from a 1D index.
308  // For example:
309  //
310  // 0 1 2 ... nbu-1
311  // nbu nbu+1 i ...
312  //
313  // We have i, we're looking for the equivalent (u=2, v=1) location.
314  // i = u + v*nbu
315  // <=> i/nbu = u/nbu + v
316  // Since 0 <= u < nbu, u/nbu = 0 (integer)
317  // Hence: v = i/nbu
318  // Then we simply put it back in the original equation to compute u = i - v*nbu
319  inline_ void Compute2DCoords(udword& u, udword& v, udword i, udword nbu)
320  {
321  v = i / nbu;
322  u = i - (v * nbu);
323  }

325  // In 3D: i = u + v*nbu + w*nbu*nbv
326  // <=> i/(nbu*nbv) = u/(nbu*nbv) + v/nbv + w
327  // u/(nbu*nbv) is null since u/nbu was null already.
328  // v/nbv is null as well for the same reason.
329  // Hence w = i/(nbu*nbv)
330  // Then we're left with a 2D problem: i' = i - w*nbu*nbv = u + v*nbu
